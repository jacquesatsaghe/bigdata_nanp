#dags/dbt_dag.py

from datetime import datetime
from pathlib import Path
from cosmos import DbtDag, ProjectConfig, ProfileConfig, ExecutionConfig
from cosmos.profiles import PostgresUserPasswordProfileMapping

profile_config = ProfileConfig(
    profile_name="default",
    target_name="dev",
    profile_mapping=PostgresUserPasswordProfileMapping(
        conn_id="postgres_default", # Créez cette connexion dans l'UI avec user/password
        profile_args={"schema": "public"},
    ),
)

dbt_postgres_dag = DbtDag(
    project_config=ProjectConfig(Path("/opt/airflow/dbt")),
    profile_config=profile_config,
    execution_config=ExecutionConfig(
        dbt_executable_path="/opt/airflow/dbt_venv/bin/dbt",
    ),
    schedule_interval="@daily",
    start_date=datetime(2024, 1, 1),
    catchup=False,
    dag_id="dbt_transformation",
)

| id | user_id | order_date | amount | status |
| :--- | :--- | :--- | :--- | :--- |

| 3 | 101 | 2024-01-02 | 20.0 | 'returned' |

# dbt_project/models/daily_revenue.sql

-- models/daily_revenue.sql

{{ config(materialized='table') }}

WITH base_orders AS (
    SELECT 
        order_date,
        amount
    FROM {{ source('public', 'raw_orders') }}
    WHERE status = 'completed'
)

SELECT
    order_date,
    SUM(amount) as total_revenue,
    COUNT(*) as order_count
FROM base_orders
GROUP BY order_date
ORDER BY order_date DESC

# Ce que dbt va faire

Lorsque Cosmos lancera ce modèle dans Airflow :

    Il va se connecter à Postgres.
    Il va créer une nouvelle table nommée daily_revenue.
    Il va exécuter le SELECT et insérer les résultats.

Résultat dans Postgres :
Table daily_revenue :
| order_date | total_revenue | order_count |
| :--- | :--- | :--- |
| 2024-01-01 | 80.0 | 2 |
