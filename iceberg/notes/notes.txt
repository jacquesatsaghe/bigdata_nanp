# nommage
tous les fichiers .properties se trouvant dans /etc/trino/catalog/
sont considérés comme base de données (file_whitout_ext.schema.table)
Example:
    /etc/trino/catalog/iceberg.properties
    CREATE SCHEMA iceberg.test_db;

    /etc/trino/catalog/polaris.properties
    CREATE SCHEMA polaris.test_db;

    /etc/trino/catalog/lakehouse.properties
    CREATE SCHEMA lakehouse.test_db;

# Se connecte directement au catalogue iceberg
docker exec -it trino trino --catalog iceberg

## use case 1 [NESSIE]: les données sont stockées dans notre warehouse
# look at nessie-bucket/warehouse/test_db/*
# file: nessie.properties
CREATE SCHEMA nessie.test_db;
CREATE TABLE nessie.test_db.data (id int, val varchar);
INSERT INTO nessie.test_db.data VALUES (1, 'Hello Nessie');

## use case 2[POLARIS]: les données sont stockées dans notre warehouse
# look at polaris-warehouse/agence_raw/*
CREATE SCHEMA IF NOT EXISTS polaris.agence_raw;

CREATE TABLE polaris.agence_raw.client (
    id BIGINT,
    code VARCHAR,
    name VARCHAR,
    actif BOOLEAN
)
WITH (
    format = 'PARQUET'
);

INSERT INTO polaris.agence_raw.client (id, code, name, actif)
VALUES (1, 'CL001', 'Jean Dupont', true);

SELECT * FROM polaris.agence_raw.client;

CREATE TABLE IF NOT EXISTS iceberg.kafka_landing.client (
    id bigint,
    code varchar,
    name varchar,
    actif int,
    year varchar,
    month varchar,
    day varchar,
    hour varchar
)
WITH (
    partitioning = ARRAY['year', 'month', 'day', 'hour']
