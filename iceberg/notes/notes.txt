# nommage
tous les fichiers .properties se trouvant dans /etc/trino/catalog/
sont considérés comme base de données (file_whitout_ext.schema.table)
Example:
    /etc/trino/catalog/iceberg.properties
    CREATE SCHEMA iceberg.test_db;

    /etc/trino/catalog/lakehouse.properties
    CREATE SCHEMA iceberg.test_db;

# Se connecte directement au catalogue iceberg
docker exec -it trino trino --catalog iceberg

## use case 1: les données sont stockées dans notre warehouse
# look at nessie-bucket/warehouse/test_db/*
CREATE SCHEMA iceberg.test_db;
CREATE TABLE iceberg.test_db.data (id int, val varchar);
INSERT INTO iceberg.test_db.data VALUES (1, 'Hello Nessie');

## use case 2: les données sont stockées dans un autre warehouse (par kafka)
# look at client-bucket/topics/fullfillment.TYROK.TYROK.client/year=2026/month=02/day=28/hour=15

# transformer un répertoire de fichiers Parquet en une table Iceberg proprement indexée sans déplacer les fichiers.
!!!!!!Procedure not registered: system.migrate
***CALL iceberg.system.migrate(
    schema_name => 'kafka_landing', 
    table_name => 'client'
);

CREATE TABLE IF NOT EXISTS iceberg.kafka_landing.client (
    id bigint,
    code varchar,
    name varchar,
    actif int,
    year varchar,
    month varchar,
    day varchar,
    hour varchar
)
WITH (
    partitioning = ARRAY['year', 'month', 'day', 'hour']
);

!!!!!! Procedure not registered: system.add_files
****CALL iceberg.system.add_files(
    schema_name => 'kafka_landing',
    table_name => 'client',
    source_path => 's3a://client-bucket/'
);


# Kafka produit souvent beaucoup de petits fichiers Parquet.
# Iceberg permet de les fusionner plus tard avec la commande
ALTER TABLE iceberg.kafka_landing.client EXECUTE optimize;
